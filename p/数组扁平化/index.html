function myflat(arr) {
    const res = [];
    const stack = [].concat(arr);
    while (stack.length !== 0) {
        const val = stack.pop();
        if (Array.isArray(val)) {
            stack.push(...val);
        } else {
            res.unshift(val);
        }
    }
    return res;
}
